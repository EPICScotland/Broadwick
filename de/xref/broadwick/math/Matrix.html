16  package broadwick.math;
17  
18  import org.apache.commons.math3.linear.Array2DRowRealMatrix;
19  import org.apache.commons.math3.linear.MatrixUtils;
20  
21  /**
22   * Implementation of real valued Matrix.
23   */
24  public class Matrix {
25  
26      /**
27       * Create a matrix from the internal representation of the matrix (copy construtor). This constructor
28       * is intentionally private to hide the internal representation of the matrix.
29       * @param data the data to be copied.
30       */
31      private Matrix(final Array2DRowRealMatrix data) {
32          this.numRows = data.getRowDimension();
33          this.numCols = data.getColumnDimension();
34          this.data = (Array2DRowRealMatrix) data.copy();
35      }
36  
37      /** 
38       * Create a matrix using the data in an array (copy constructor). This constructor will copy the contents
39       * of the argument.
40       * @param data the data array to be copied.
41       */
42      private Matrix(final double[][] data) {
43          this.data = new Array2DRowRealMatrix(data);
44          this.numRows = this.data.getRowDimension();
45          this.numCols = this.data.getColumnDimension();
46      }
47  
48      /**
49       * Create a matrix with the given dimensions. 
50       * @param rowDimension the number of rows in the matrix.
51       * @param columnDimension the number of columns in the matrix.
52       */
53      public Matrix(final int rowDimension, final int columnDimension) {
54          this.numRows = rowDimension;
55          this.numCols = columnDimension;
56          data = new Array2DRowRealMatrix(rowDimension, columnDimension);
57  
58          for (int i = 0; i < rowDimension; i++) {
59              for (int j = 0; j < columnDimension; j++) {
60                  data.setEntry(i, j, 0.0);
61              }
62          }
63      }
64  
65      /**
66       * Get the contents of the matrix. The (i,j) element of the matrix is returned where i is the row index and j 
67       * is the column index.
68       * @param row the row index.
69       * @param column the column index.
70       * @return the entry at eh requiered index.
71       */
72      public double element(final int row, final int column) {
73          return data.getEntry(row, column);
74      }
75  
76      /**
77       * Set the matrix element at a given point.
78       * @param row the row index of the new element.
79       * @param column the column element of the new element.
80       * @param value the value to be set.
81       * @return the new matrix element at the given coordinates.
82       */
83      public double setEntry(final int row, final int column, final double value) {
84          data.setEntry(row, column, value);
85          return data.getEntry(row, column);
86      }
87  
88      /**
89       * Return an array representation of this matrix.
90       * @return a double[][] containing the values of this matrix.
91       */
92      public double[][] toArray() {
93          return data.getData();
94      }
95  
96      /**
97       * Create a copy of the matrix. A new matrix is returned with the same contents of this one.
98       * @return a copy of this matrix.
99       */
100     public Matrix copy() {
101         return new Matrix(data);
102     }
103 
104     /**
105      * Get the number of rows in the matrix.
106      * @return the number of rows.
107      */
108     public int rows() {
109         return numRows;
110     }
111 
112     /**
113      * Get the number of columns in the matrix.
114      * @return the number of columns.
115      */
116     public int columns() {
117         return numCols;
118     }
119 
120     @Override
121     public String toString() {
122         StringBuilder sb = new StringBuilder();
123 
124         for (int i = 0; i < numRows; i++) {
125             for (int j = 0; j < numCols; j++) {
126                 sb.append(data.getEntry(i, j)).append(",");
127             }
128             sb.append("\n");
129         }
130         return sb.toString();
131     }
132 
133     /**
134      * Get the inverse of the matrix. This will return a new matrix object that is the inverse.
135      * @return a Matrix object that is the inverse of this matrix.
136      */
137     public Matrix inverse() {
138         return new Matrix(MatrixUtils.inverse(data).getData());
139     }
140 
141     /**
142      * Get the transpose of the matrix. This will return a new matrix object that is the transpose.
143      * @return a Matrix object that is the transpose of this matrix.
144      */
145     public Matrix transpose() {
146         return new Matrix(data.transpose().getData());
147     }
148 
149     /**
150      * Returns the result of subtracting m from this.
151      * @param m matrix to be subtracted
152      * @return this-m
153      */
154     public Matrix subtract(final Matrix m) {
155         return new Matrix(data.subtract(m.data));
156     }
157 
158     /**
159      * Returns the result of postmultiplying this by m.
160      * @param m matrix to postmultiply by
161      * @return this*m
162      */
163     public Matrix multiply(final Matrix m) {
164         return new Matrix(data.multiply(m.data));
165     }
166     
167         /**
168      * Returns the result of postmultiplying this by a scalar.
169      * @param d the scalar with which we will multiply this matrix
170      * @return this*d
171      */
172     public Matrix multiply(final double d) {
173         return new Matrix(data.scalarMultiply(d).getData());
174     }
175 
176     /**
177      * Returns the result of multiplying this by the vector v.
178      * @param v the vector.
179      * @return this*v
180      */
181     public Vector multiply(final Vector v) {
182         return new Vector(data.operate(v.toArray()));
183     }
184 
185     /**
186      * Returns the (row) vector result of premultiplying this by the vector v.
187      * @param v the row vector to premultiply by
188      * @return v*this
189      */
190     public Vector preMultiply(final Vector v) {
191         return new Vector(data.preMultiply(v.toArray()));
192     }
193 
194     private final int numRows;
195     private final int numCols;
196     private final Array2DRowRealMatrix data;
197 }
