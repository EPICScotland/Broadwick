36   * Base class for all data file readers, containing useful common functionality..

41      /**
42       * Convert a collection of elements read from the data file (the column names in the internal database) into a CSV
43       * string.
44       * @param data the collection of column names.
45       * @return a csv string of the input data.
46       */
47      protected final String asCsv(final Collection<String> data) {
48          final StringBuilder sb = new StringBuilder();
49          for (String name : data) {
50              if (name != null) {
51                  sb.append(name).append(",");
52              }
53          }
54          // remove the last character.
55          final String csv = sb.toString();
56          return csv.substring(0, csv.length() - 1);
57      }
58  
59      /**
60       * Convert a collection of elements read from the data file (the column names in the internal database) into a CSV
61       * string of question marks.
62       * @param data the collection of column names.
63       * @return as csv string of the question marks containing the same number of data elements.
64       */
65      protected final String asQuestionCsv(final Collection<String> data) {
66          final StringBuilder sb = new StringBuilder();
67          for (String name : data) {
68              if (name != null) {
69                  sb.append("?,");
70              }
71          }
72          // remove the last character.
73          final String csv = sb.toString();
74          return csv.substring(0, csv.length() - 1);
75      }

77      /**
78       * Create or append to a string command in the form "CREATE TABLE IF NOT EXISTS [table] ([col1], [col2], ...,
79       * [coln]) values (?,?,...,?)" from the details read from the configuration file.
80       * <pre>
81       * updateCreateTableCommand(ID, 1, " VARCHAR(128), ", new HashMap<String, Integer>(), createTableCommand,
82       * TABLE_NAME, SECTION_NAME, errors);
83       * </pre>
84       * @param columnName         the name of the column in the table
85       * @param columnIndex        the location in the csv file where it appears.
86       * @param columnType         the type of column e.g. "INT", "VARCHAR(128) NOT NULL" etc.
87       * @param insertedColDetails a map containing the name of the column and the location in the csv file where it
88       *                           appears i.e. columnName, columnIndex.
89       * @param createTableCommand a StringBuilder object that we append to to create the create table command.
90       * @param tableName          the name of the table that is being created.
91       * @param sectionName        the name of the section in the configuration file that configures the data.
92       * @param errors             a stringbuilder object to which we will append error messages.
93       * @return the name of the column added.
94       */
95      protected final String updateCreateTableCommand(final String columnName, final Integer columnIndex, final String columnType,
96                                                      final Map<String, Integer> insertedColDetails,
97                                                      final StringBuilder createTableCommand, final String tableName,
98                                                      final String sectionName, final StringBuilder errors) {
99          if (createTableCommand.length() == 0) {
100             createTableCommand.append(String.format("CREATE TABLE %s (", tableName));
101         }
102 
103         String addedColumnName = null;
104         if (columnIndex != null) {
105             if (columnIndex != 0) {
106                 createTableCommand.append(columnName).append(columnType);
107                 insertedColDetails.put(columnName, columnIndex);
108                 addedColumnName = columnName;
109             } else {
110                 errors.append("No ").append(columnName).append(" column set in ").append(sectionName).append(" section\n");
111             }
112         }
113 
114         return addedColumnName;
115     }

117     /**
118      * Execute a command to create a table.
119      * @param tableName          the name of the table to be created.
120      * @param createTableCommand the command to create the table.
121      * @param connection         the database connection to use to create the table.
122      * @throws SQLException if a SQL error has been encountered.
123      */
124     protected final void createTable(final String tableName, final String createTableCommand, final Connection connection)
125             throws SQLException {
126 
127         final DatabaseMetaData dbm = connection.getMetaData();
128 
129         // First check if the table already exists, some databases do not support
130         // CREATE TABLE ??? IF NOT EXISTS
131         // so we have to look at the database schema
132         try (ResultSet resultSet = dbm.getTables(null, null, "%", null)) {
133             boolean tableExists = false;
134             while (resultSet.next()) {
135                 if (tableName.equalsIgnoreCase(resultSet.getString("TABLE_NAME"))) {
136                     log.debug("Table {} already exists, ignoring", tableName);
137                     tableExists = true;
138                 }
139             }
140 
141             if (!tableExists) {
142                 try (Statement stmt = connection.createStatement()) {
143                     final String[] commands = createTableCommand.split(";");
144                     for (int i = 0; i < commands.length; i++) {
145                         log.trace("Creating table {}", commands[i]);
146                         stmt.execute(commands[i]);
147                     }
148                 } catch (SQLException sqle) {
149                     connection.rollback();
150                     log.error("Error while creating the table '{}'. {}", createTableCommand,
151                               Throwables.getStackTraceAsString(sqle));
152                     throw sqle;
153                 }
154             }
155 
156 //        } catch (Exception e) {
157 //            log.error("Could not create database {}", Throwables.getStackTraceAsString(e));
158         }
159 
160         connection.commit();
161     }

163     /**
164      * Perform the insertion into the database.
165      * @param connection      the connection to the database.
166      * @param tableName       the name of the table into which the data will be put.
167      * @param insertString    the command used to insert a row into the database.
168      * @param dataFile        the [CSV] file that contained the data.
169      * @param dateFormat      the format of the date in the file.
170      * @param insertedColInfo a map of column name to column in the data file.
171      * @param dateFields      a collection of columns in the csv file that contains date fields.
172      * @return the number of rows inserted.
173      */
174     protected final int insert(final Connection connection, final String tableName,
175                                final String insertString,
176                                final String dataFile,
177                                final String dateFormat,
178                                final Map<String, Integer> insertedColInfo,
179                                final Collection<Integer> dateFields) {
180 
181         int inserted = 0;
182         try {
183             // Now do the insertion.
184             log.trace("Inserting into {} via {}", tableName, insertString);
185             PreparedStatement pstmt = connection.prepareStatement(insertString);
186             log.trace("Prepared statement = {}", pstmt.toString());
187 
188             try (FileInput instance = new FileInput(dataFile, ",")) {
189                 final StopWatch sw = new StopWatch();
190                 sw.start();
191                 List<String> data = instance.readLine();
192                 while (data != null && !data.isEmpty()) {
193                     int parameterIndex = 1;
194                     for (Map.Entry<String, Integer> entry : insertedColInfo.entrySet()) {
195                         if (entry.getValue() == -1) {
196                             pstmt.setObject(parameterIndex, null);
197                         } else {
198                             final String value = data.get(entry.getValue() - 1);
199                             if (dateFields.contains(entry.getValue())) {
200                                 int dateField = Integer.MAX_VALUE;
201                                 if (value != null && !value.isEmpty()) {
202                                     dateField = BroadwickConstants.getDate(value, dateFormat);
203                                 }
204                                 pstmt.setObject(parameterIndex, dateField);
205                             } else {
206                                 pstmt.setObject(parameterIndex, value);
207                             }
208                         }
209                         parameterIndex++;
210                     }
211                     pstmt.addBatch();
212                     try {
213                         pstmt.executeUpdate();
214                         inserted++;
215                     } catch (SQLException ex) {
216                         if ("23505".equals(ex.getSQLState())) {
217                             //Ignore found duplicate from database view
218                             continue;
219                         } else {
220                             log.warn("Duplicate data found for {}: continuing despite errors: {}", data.get(0), ex.getLocalizedMessage());
221                             log.trace("{}", Throwables.getStackTraceAsString(ex));
222                             throw ex;
223                         }
224                     }
225                     if (inserted % 250000 == 0) {
226                         log.trace("Inserted {} rows in {}", inserted, sw.toString());
227                         connection.commit();
228                         pstmt.close();
229                         pstmt = connection.prepareStatement(insertString);
230                     }
231 
232                     data = instance.readLine();
233                 }
234                 connection.commit();
235 
236             } catch (IOException ex) {
237                 log.error("IO error : {}", ex.getLocalizedMessage());
238                 log.trace("{}", Throwables.getStackTraceAsString(ex));
239             } catch (SQLException ex) {
240                 log.error("SQL Error : {}", ex.getLocalizedMessage());
241                 log.trace("{}", Throwables.getStackTraceAsString(ex));
242                 throw ex;
243             } finally {
244                 pstmt.close();
245             }
246         } catch (SQLException ex) {
247             log.error("{}", ex.getLocalizedMessage());
248             log.trace("{}", Throwables.getStackTraceAsString(ex));
249             throw new BroadwickException(ex);
250         }
251 
252         return inserted;
253     }
