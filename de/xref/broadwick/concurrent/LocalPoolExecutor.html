16  package broadwick.concurrent;

18  import broadwick.BroadwickException;
19  import com.google.common.base.Throwables;
20  import java.util.ArrayList;
21  import java.util.Iterator;
22  import java.util.List;
23  import java.util.concurrent.Callable;
24  import java.util.concurrent.CountDownLatch;
25  import java.util.concurrent.ExecutionException;
26  import java.util.concurrent.ExecutorService;
27  import java.util.concurrent.Executors;
28  import java.util.concurrent.Future;
29  import java.util.concurrent.TimeUnit;
30  import lombok.Getter;
31  import lombok.extern.slf4j.Slf4j;

33  /**
34   * This Executor controls the execution of several threads using a fixed thread pool. Several instances of the callable
35   * object are added to the execution pool which monitors their execution, if any fails, i.e. any of the jobs throws an
36   * exception, then all non-running jobs are canceled and currently running jobs are terminated.
37   */
38  @Slf4j
39  public class LocalPoolExecutor implements Executor {
40  
41      /**
42       * Create the Callable executor that will run a supplied number of instances of a supplied callable, canceling the execution
43       * if a thread throws an exception.
44       * @param threadPoolSize    the number of concurrent threads to be run (the size of the thread pool)
45       * @param numInstancesToRun the number of threads to be run.
46       * @param callable          the actual Callable instance that is to be executed in the pool.
47       */
48      public LocalPoolExecutor(final int threadPoolSize, final int numInstancesToRun, final Callable<?> callable) {
49          poolSize = threadPoolSize;
50          numRuns = numInstancesToRun;
51          job = callable;
52          service = Executors.newFixedThreadPool(poolSize);
53      }
54  
55      /**
56       * Create the Callable executor that will run each callable in a collection, canceling the execution
57       * if a thread throws an exception.
58       * @param threadPoolSize    the number of concurrent threads to be run (the size of the thread pool)
59       * @param jobs          a collection of [Callable] jobs to be run.
60       */
61      public LocalPoolExecutor(final int threadPoolSize, final List<Callable<?>> jobs) {
62          poolSize = threadPoolSize;
63          numRuns = jobs.size();
64          job = null;
65          service = Executors.newFixedThreadPool(poolSize);
66      }
67  
68      @Override
69      public final Status getStatus() {
70          return status;
71      }
72  
73      @Override
74      public final void run() {
75  
76          final List<Future<?>> tasks = new ArrayList<>();
77          final CountDownLatch latch = new CountDownLatch(numRuns);
78  
79          status = Executor.Status.RUNNING;
80  
81          // Submit all the Callable objects to the executor, counting down the latch
82          // when each submitted job is finished. If any of the jobs throws an exception,
83          // it will be wrapped in an ExecutionException by the Future and we will check for it later.
84          for (int i = 0; i < numRuns; i++) {
85              final int taskId = i;
86              log.trace("Adding task {} to the executor service.", i);
87              tasks.add(service.submit(new Runnable() {
88                  @Override
89                  public void run() {
90                      try {
91                          job.call();
92                      } catch (Exception e) {
93                          ok = false;
94                          throw new BroadwickException(e);
95                      } finally {
96                          log.debug("Completed task {}", taskId);
97                          latch.countDown();
98                      }
99                  }
100             }));
101         }
102 
103         // We now have all the jobs submitted to the executor service. Now we iteratively check the status of each 
104         // job (sleeping for 100ms between checks) to determine if each job has been completed. If any job
105         // fails then we catch the resultant ExecutionException and cancel all remaining jobs.
106         int completedTasks = 0;
107         while (completedTasks < tasks.size()) {
108             try {
109                 final Iterator<Future<?>> iterator = tasks.iterator();
110                 while (iterator.hasNext()) {
111                     final Future<?> task = iterator.next();
112                     if (task.isDone()) {
113                         completedTasks++;
114                         task.get();
115                         iterator.remove();
116                     } else if (task.isCancelled()) {
117                         stopAllTasks(tasks);
118                     }
119                 }
120                 Thread.sleep(10);
121             } catch (InterruptedException e) {
122                 Thread.currentThread().interrupt();
123                 status = Executor.Status.TERMINATED;
124                 ok = false;
125                 break;
126             } catch (ExecutionException e) {
127                 // One of the submitted jobs threw an exception, we need to stop all the currently running threads.
128                 log.error("LocalPoolExecutor detected an error: {}", Throwables.getStackTraceAsString(e));
129                 stopAllTasks(tasks);
130                 service.shutdown();
131                 status = Executor.Status.TERMINATED;
132                 ok = false;
133                 break;
134             }
135         }
136 
137         try {
138             if (status != Executor.Status.TERMINATED) {
139                 latch.await();
140             }
141         } catch (InterruptedException e) {
142             log.error("Caught exception closing failed tasks");
143             status = Executor.Status.TERMINATED;
144 
145         }
146 
147         shutdown();
148     }
149 
150     /**
151      * Send a message to all the futures to cancel their execution. On completion the status of the executor is set to
152      * Executor.Status.TERMINATED.
153      * @param tasks a collection of future object that are to be stopped.
154      */
155     private void stopAllTasks(final List<Future<?>> tasks) {
156         for (Future<?> task : tasks) {
157             task.cancel(true);
158         }
159         status = Executor.Status.TERMINATED;
160     }
161 
162     /**
163      * Shutdown the ExecutionService, blocking until ALL the threads have finished before returning.
164      */
165     private void shutdown() {
166         log.trace("Shutting down the LocalPoolExecutor");
167         try {
168             service.shutdown();
169             service.awaitTermination(1000, TimeUnit.MILLISECONDS);
170 
171             log.trace("LocalPoolExecutor succesfully shutdown");
172             status = Executor.Status.COMPLETED;
173 
174         } catch (InterruptedException e) {
175             // Nothing to do - we are shuting done the service
176             Thread.currentThread().interrupt();
177             status = Executor.Status.TERMINATED;
178         } catch (Exception e) {
179             // Nothing to do - we are shuting done the service
180             status = Executor.Status.TERMINATED;
181         }
182     }
183     private final int poolSize;
184     private final int numRuns;
185     private final Callable<?> job;
186     private final ExecutorService service;
187     private Status status = Executor.Status.NOT_STARTED;
188     @Getter
189     @SuppressWarnings("PMD.UnusedPrivateField")
190     private boolean ok = true; /// true if the underlying task completed without error, false otherwise
191 }
