16  package broadwick.statistics.regression;
17  
18  import org.apache.commons.math3.stat.regression.SimpleRegression;
19  
20  /**
21   * Estimates an ordinary least squares regression model with one independent variable. y = intercept + slope * x
22   */
23  public class SimpleLinearRegression {
24  
25      /**
26       * Create an empty SimpleLinearRegression instance.
27       */
28      public SimpleLinearRegression() {
29          regression = new SimpleRegression();
30      }
31  
32      /**
33       * Adds the observation (x,y) to the regression data set. Uses updating formulas for means and sums of squares
34       * defined in "Algorithms for Computing the Sample Variance: Analysis and Recommendations", Chan, T.F., Golub, G.H.,
35       * and LeVeque, R.J. 1983, American Statistician, vol. 37, pp. 242-247, referenced in Weisberg, S. "Applied Linear
36       * Regression". 2nd Ed. 1985. Note: this uses the apache commons math library.
37       * @param x independent variable value
38       * @param y dependent variable value
39       */
40      public final void addData(final double x, final double y) {
41          regression.addData(x, y);
42      }
43      
44      
45      /**
46       * Removes the observation (x,y) from the regression data set, mirroring the addData method. 
47       * The method has no effect if there are no points of data (i.e. n=0)
48       * @param x independent variable value
49       * @param y dependent variable value
50       */
51      public final void removeData(final double x, final double y) {
52          regression.removeData(x, y);
53      }
54  
55      /**
56       * Returns the slope of the estimated regression line. At least two observations (with at least two different x
57       * values) must have been added before invoking this method. If this method is invoked before a model can be
58       * estimated,
59       * <code>Double,NaN</code> is returned.
60       * @return the slope of the regression line
61       */
62      public final double getSlope() {
63          return regression.getSlope();
64      }
65  
66      /**
67       * Returns the intercept of the estimated regression line. At least two observations (with at least two different x
68       * values) must have been added before invoking this method. If this method is invoked before a model can be
69       * estimated,
70       * <code>Double,NaN</code> is returned.
71       * @return the intercept of the regression line
72       */
73      public final double getIntercept() {
74          return regression.getIntercept();
75      }
76  
77      /**
78       * Returns the number of observations that have been added to the model.
79       * @return n number of observations that have been added.
80       */
81      public final long getN() {
82          return regression.getN();
83      }
84  
85      /**
86       * Returns the "predicted"
87       * <code>y</code> value associated with the supplied
88       * <code>x</code> value, based on the data that has been added to the model when this method is activated.
89       * <p>
90       * <code> predict(x) = intercept + slope * x </code>
91       * </p>
92       * At least two observations (with at least two different x values) must have been added before invoking this
93       * method. If this method is invoked before a model can be estimated,
94       * <code>Double,NaN</code> is returned.
95       * @param x input <code>x</code> value
96       * @return predicted <code>y</code> value
97       */
98      public final double predict(final double x) {
99          return regression.predict(x);
100     }
101 
102     /**
103      * Returns <a href="http://mathworld.wolfram.com/CorrelationCoefficient.html"> Pearson's product moment correlation
104      * coefficient</a>, usually denoted r. At least two observations (with at least two different x values) must have
105      * been added before invoking this method. If this method is invoked before a model can be estimated,
106      * <code>Double,NaN</code> is returned.
107      * <p/>
108      * @return Pearson's r
109      */
110     public final double getR() {
111         return regression.getR();
112     }
113 
114     /**
115      * Returns the <a href="http://www.xycoon.com/standarderrorb0.htm">
116      * standard error of the intercept estimate</a>, usually denoted s(b0). If there are fewer that
117      * <strong>three</strong> observations in the model, or if there is no variation in x, this returns
118      * <code>Double.NaN</code>.
119      * @return standard error associated with intercept estimate
120      */
121     public final double getInterceptStdErr() {
122         return regression.getInterceptStdErr();
123     }
124 
125     /**
126      * Returns the <a href="http://www.xycoon.com/standerrorb(1).htm">standard error of the slope estimate</a>. If there
127      * are fewer that <strong>three</strong> data pairs in the model, or if there is no variation in x, this returns
128      * <code>Double.NaN</code>.
129      * @return standard error associated with slope estimate
130      */
131     public final double getSlopeStdErr() {
132         return regression.getSlopeStdErr();
133     }
134     
135     @Override
136     public final String toString() {
137         return String.format("Slope [%f], Intercept [%f]. N=%d", getSlope(), getIntercept(), getN());
138     }
139     private SimpleRegression regression;
140 }
