16  package broadwick.stochastic;
17  
18  import java.util.Collection;
19  import lombok.EqualsAndHashCode;
20  import lombok.Getter;
21  
22  /**
23   * Implementing classes are used to observe certain aspects of simulations (stochastic processes). It is basically just
24   * an interface of methods that get called during a simulation together with some common methods useful for many types
25   * of observers.
26   * <p/>
27   * Some observers take repeats into account. If you repeat the simulations, they will yield average data.
28   * <p/>
29   * Each observer should override the
30   * <code>toString</code> method which should return a string representation of its data.
31   * <p/>
32   * You can register a {@link PrintWriter} to the
33   * <code>Observer</code>. If you have, the return value of
34   * <code>toString</code> is printed if a simulation has finished.
35   * <p/>
36   * @author Florian Erhard (copied from FERN http://www.bio.ifi.lmu.de/FERN/)
37   * <p/>
38   */
39  @EqualsAndHashCode
40  @SuppressWarnings("squid:S1210")
41  public abstract class Observer implements Comparable<Observer> {
42  
43      /**
44       * Creates an observer dedicated to one process. The observer is NOT registered at the process, you have to call
45       * {@link Simulator#addObserver(Observer)} in order to do this.
46       * <p/>
47       * @param sim the process
48       */
49      public Observer(final StochasticSimulator sim) {
50          this.process = sim;
51      }
52  
53      @Override
54      public final int compareTo(final Observer o) {
55          // We really are only interested in determining if the Observers are equal so that a tree based table 
56          // can distinguish between observers. Order does NOT matter so we return 1 if the objects are not equal.
57          if (this == o) {
58              return 0;
59          }
60          return 1;
61      }
62      
63      /**
64       * Gets called when the simulation has started after the initialization and before the termination condition is
65       * checked the first time.
66       */
67      public abstract void started();
68  
69      /**
70       * Gets called after each termination check and before
71       * {@link Simulator#performStep(fern.simulation.controller.SimulationController)} is called.
72       */
73      public abstract void step();
74  
75      /**
76       * Gets called when a simulation has finished, directly after the termination check.
77       */
78      public abstract void finished();
79  
80      /**
81       * Gets called by simulators when a certain moment in time is reached. This moment in time has to be registered with
82       * the observer in the StochasticSimulator.
83       * @param thetaTime moment in time
84       * @param events    a collection of events that occur at thetaTime.
85       */
86      public abstract void theta(final double thetaTime, final Collection<Object> events);
87  
88      /**
89       * Gets called before an event is triggered.
90       * @param event the event which is supposed to fire
91       * @param tau   the time the event occurs (at this time {@link Simulator#getTime()}
92       * @param times the number of firings
93       */
94      public abstract void observeEvent(final SimulationEvent event, final double tau, final int times);
95  
96      @Getter
97      @SuppressWarnings("PMD.UnusedPrivateField")
98      private StochasticSimulator process;
99  }
