16  package broadwick.stochastic.algorithms;
17  
18  import broadwick.rng.RNG;
19  import broadwick.stochastic.AmountManager;
20  import broadwick.stochastic.SimulationEvent;
21  import broadwick.stochastic.StochasticSimulator;
22  import broadwick.stochastic.TransitionKernel;
23  import java.util.LinkedHashSet;
24  import java.util.Set;
25  import lombok.extern.slf4j.Slf4j;
26  
27  /**
28   * Abstract base class for tau leaping classes.
29   */
30  @Slf4j
31  public abstract class AbstractTauLeapingBase extends StochasticSimulator {
32  
33      /**
34       * Create the tau-leaping object.
35       * @param amountManager the amount manager used in the simulator.
36       * @param transitionKernel the transition kernel to be used with the stochastic solver.
37       */
38      public AbstractTauLeapingBase(final AmountManager amountManager, final TransitionKernel transitionKernel) {
39          super(amountManager, transitionKernel);
40      }
41  
42      /**
43       * Update all the reactions/transitions that have occurred in the tau step.
44       * @param tau the step size.
45       */
46      protected final void updateStates(final double tau) {
47          log.trace("AbstractTauLeapingBase: Updating states at {} using a step size of {}", getCurrentTime(), tau);
48          final Set<SimulationEvent> transitionEvents = new LinkedHashSet<>();
49          transitionEvents.addAll(getTransitionKernel().getTransitionEvents());
50          
51          for (SimulationEvent event : transitionEvents) {
52              if (getTransitionKernel().getTransitionEvents().contains(event)) {
53                  final double rate = getTransitionKernel().getTransitionProbability(event);
54                  final double times = GENERATOR.getPoisson(rate * tau);
55                  if (times > 0) {
56                      log.trace("Updating states {}", new StringBuilder()
57                              .append("for event ").append(event.toString())
58                              .append(", rate = ").append(rate)
59                              .append(", times = ").append(times)
60                              .toString());
61                      doEvent(event, getCurrentTime(), (int) Math.round(times));
62                      log.trace("fired event {} {} time(s)", event, times);
63                  }
64              }
65          }
66      }
67  
68      /**
69       * Determine of there is a theta event in the current time step, taking into
70       * account possible reverse time jumps.
71       * @param tau the time interval (will be negative for reverse time).
72       * @return true if a theta event is found.
73       */
74      protected final boolean isThetaEventInCurrentStep(final double tau) {
75          final double nextEventTime = getNextThetaEventTime();
76          if (reverseTime) {
77              return getCurrentTime() >= nextEventTime && getCurrentTime() + tau < nextEventTime;
78          } else {
79              return getCurrentTime() <= nextEventTime && getCurrentTime() + tau > nextEventTime;
80          }
81      }
82      protected static final RNG GENERATOR = new RNG(RNG.Generator.Well19937c);
83  }
