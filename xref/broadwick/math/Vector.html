16  package broadwick.math;
17  
18  /**
19   * Implementation of mathematical vector (not to be confused with the data structure). It is perhaps best to think of
20   * this as a row-vector as several underlying operations (such as multiplying by a matrix) are implemented by casting
21   * the vector into a 1D (row) matrix which can be transposed etc internally.
22   */
23  public class Vector {
24  
25      /**
26       * Create a vector of a given dimension. Each element will be set to 0.
27       * @param dimension the dimension of the vector.
28       */
29      public Vector(final int dimension) {
30          data = new double[dimension];
31  
32          for (int i = 0; i < dimension; i++) {
33              data[i] = 0.0;
34          }
35      }
36  
37      /**
38       * Copy constructor.
39       * @param v the vector to be copied.
40       */
41      public Vector(final Vector v) {
42          this(v.data);
43      }
44  
45      /**
46       * Copy constructor.
47       * @param data an array that will be copied.
48       */
49      public Vector(final double[] data) {
50          this.data = new double[data.length];
51  
52          System.arraycopy(data, 0, this.data, 0, data.length);
53      }
54  
55      /**
56       * Get the element at the given index.
57       * @param i the vector element to be found.
58       * @return the vector element
59       */
60      public double element(final int i) {
61          return data[i];
62      }
63  
64      /**
65       * Set the value of the vector at a given index.
66       * @param i     the coordinate of the vector to be set
67       * @param value the value of the vector element.
68       * @return the updated value.
69       */
70      public double setEntry(final int i, final double value) {
71          data[i] = value;
72          return data[i];
73      }
74  
75      /**
76       * Return a copy of the vector as an array.
77       * @return a copy of the vector as an array.
78       */
79      public double[] toArray() {
80          return data.clone();
81      }
82  
83      /**
84       * Obtain the length, i.e. the number of components of this vector.
85       * @return the number of elements in the vector.
86       */
87      public int length() {
88          return data.length;
89      }
90  
91      /**
92       * Add a vector to this one.
93       * @param v the vector to be added to this one.
94       * @return this (updated) vector.
95       */
96      public Vector add(final Vector v) {
97          Vector v1 = new Vector(data);
98  
99          for (int i = 0; i < data.length; i++) {
100             v1.setEntry(i, v1.element(i) + v.element(i));
101         }
102         return v1;
103     }
104 
105     /**
106      * Subtract a vector from this one.
107      * @param v the vector to be subtracted from this one.
108      * @return this (updated) vector.
109      */
110     public Vector subtract(final Vector v) {
111         Vector v1 = new Vector(data);
112 
113         for (int i = 0; i < data.length; i++) {
114             v1.setEntry(i, v1.element(i) - v.element(i));
115         }
116         return v1;
117     }
118 
119     /**
120      * Calculate the dot product of this vector with v.
121      * @param v the vector to be dotted.
122      * @return this (updated) vector.
123      */
124     public double multiply(final Vector v) {
125         double d = 0.0;
126 
127         for (int i = 0; i < data.length; i++) {
128             d += data[i] * v.element(i);
129         }
130         return d;
131     }
132 
133     @Override
134     public String toString() {
135         StringBuilder sb = new StringBuilder("(");
136 
137         for (int i = 0; i < data.length; i++) {
138             sb.append(data[i]).append(",");
139         }
140         sb.deleteCharAt(sb.length() - 1);
141         sb.append(")");
142         return sb.toString();
143     }
144 
145     private final double[] data;
146 }
