18  import broadwick.BroadwickException;
19  import broadwick.config.generated.CustomTags;
20  import broadwick.config.generated.DataFiles;

22  import com.google.common.base.Throwables;
23  import java.sql.Connection;
24  import java.util.Collection;
25  import java.util.HashSet;
26  import java.util.Map;
27  import java.util.TreeMap;
28  import lombok.Getter;
29  import lombok.extern.slf4j.Slf4j;

31  /**
32   * Read a file containing the results of tests.
33   */
34  @Slf4j
35  public class TestsFileReader extends DataFileReader {
36  
37      /**
38       * Create the tests file reader.
39       * @param testsFile the [xml] tag of the config file that is to be read.
40       * @param dbImpl    the implementation of the database object.
41       */
42      public TestsFileReader(final DataFiles.TestsFile testsFile,
43                             final DatabaseImpl dbImpl) {
44  
45          super();
46          this.database = dbImpl;
47          this.dataFile = testsFile.getName();
48          this.dateFields = new HashSet<>();
49          this.dateFormat = testsFile.getDateFormat();
50          this.insertedColInfo = new TreeMap<>();
51          final StringBuilder errors = new StringBuilder();
52  
53          this.createTableCommand = new StringBuilder();
54          updateCreateTableCommand(ID, testsFile.getIdColumn(), " VARCHAR(128), ",
55                                   insertedColInfo, createTableCommand, TABLE_NAME, SECTION_NAME, errors);
56          updateCreateTableCommand(GROUP_ID, testsFile.getGroupIdColumn(), " VARCHAR(128), ",
57                                   insertedColInfo, createTableCommand, TABLE_NAME, SECTION_NAME, errors);
58          updateCreateTableCommand(LOCATION_ID, testsFile.getLocationIdColumn(), " VARCHAR(128), ",
59                                   insertedColInfo, createTableCommand, TABLE_NAME, SECTION_NAME, errors);
60          updateCreateTableCommand(TEST_DATE, testsFile.getTestDateColumn(), " INT, ",
61                                   insertedColInfo, createTableCommand, TABLE_NAME, SECTION_NAME, errors);
62          updateCreateTableCommand(POSITIVE_RESULT, testsFile.getPostiveResultColumn(), " INT, ",
63                                   insertedColInfo, createTableCommand, TABLE_NAME, SECTION_NAME, errors);
64          updateCreateTableCommand(NEGATIVE_RESULT, testsFile.getNegativeResultColumn(), " INT, ",
65                                   insertedColInfo, createTableCommand, TABLE_NAME, SECTION_NAME, errors);
66  
67          if (testsFile.getTestDateColumn() > 0) {
68              dateFields.add(testsFile.getTestDateColumn());
69          }
70  
71          if (testsFile.getCustomTags() != null) {
72              for (CustomTags.CustomTag tag : testsFile.getCustomTags().getCustomTag()) {
73                  updateCreateTableCommand(tag.getName(), tag.getColumn(), " VARCHAR(128), ",
74                                           insertedColInfo, createTableCommand, TABLE_NAME, SECTION_NAME, errors);
75                  if ("date".equals(tag.getType())) {
76                      dateFields.add(tag.getColumn());
77                  }
78              }
79          }
80  
81          final StringBuilder createIndexCommand = new StringBuilder();
82          String primaryKeyId = ID;
83          if (testsFile.getIdColumn() != null) {
84              primaryKeyId = ID;
85          } else if (testsFile.getGroupIdColumn() != null) {
86              primaryKeyId = GROUP_ID;
87          } else if (testsFile.getLocationIdColumn() != null) {
88              primaryKeyId = LOCATION_ID;
89          }
90  //        // we arent adding a primary key. 
91  //        createTableCommand.append(String.format("PRIMARY KEY (%s,%s,%s,%s) );", primaryKeyId, TEST_DATE, POSITIVE_RESULT, NEGATIVE_RESULT)); 
92  
93          createTableCommand.deleteCharAt(createTableCommand.length() - 1);
94          createTableCommand.append(");");
95  
96          createIndexCommand.append(String.format(" CREATE INDEX IF NOT EXISTS IDX_TEST_ID ON %s (%s);",
97                                                  TABLE_NAME, primaryKeyId));
98          createIndexCommand.append(String.format(" CREATE INDEX IF NOT EXISTS IDX_TEST_POS ON %s (%s,%s,%s);",
99                                                  TABLE_NAME, primaryKeyId, TEST_DATE, POSITIVE_RESULT));
100         createIndexCommand.append(String.format(" CREATE INDEX IF NOT EXISTS IDX_TEST_ALL ON %s (%s,%s,%s,%s);",
101                                                 TABLE_NAME, primaryKeyId, TEST_DATE, POSITIVE_RESULT, NEGATIVE_RESULT));
102 
103         createTableCommand.append(createIndexCommand.toString());
104 
105         insertString = String.format("INSERT INTO %s (%s) VALUES (%s)",
106                                      TABLE_NAME, asCsv(insertedColInfo.keySet()), asQuestionCsv(insertedColInfo.keySet()));
107 
108         if (errors.length() > 0) {
109             log.error(errors.toString());
110             throw new BroadwickException(errors.toString());
111         }
112     }
113 
114     @Override
115     public final int insert() {
116         log.trace("TestsFileReader insert");
117 
118         int inserted = 0;
119         try (Connection connection = database.getConnection()) {
120             createTable(TABLE_NAME, createTableCommand.toString(), connection);
121 
122             inserted = insert(connection, TABLE_NAME, insertString, dataFile, dateFormat, insertedColInfo, dateFields);
123         } catch (Exception ex) {
124             log.error("Error reading test data {}. {}", ex.getLocalizedMessage(), Throwables.getStackTraceAsString(ex));
125             throw new BroadwickException(ex);
126         }
127         return inserted;
128     }
129 
130     private DatabaseImpl database;
131     private String dataFile;
132     private String dateFormat;
133     @Getter
134     private static final String TABLE_NAME = "Tests";
135     private StringBuilder createTableCommand;
136     private String insertString;
137     private Map<String, Integer> insertedColInfo;
138     private Collection<Integer> dateFields;
139     @Getter
140     private static final String ID = "ID";
141     @Getter
142     private static final String GROUP_ID = "GROUPID";
143     @Getter
144     private static final String LOCATION_ID = "LOCATIONID";
145     @Getter
146     private static final String TEST_DATE = "TESTDATE";
147     @Getter
148     private static final String POSITIVE_RESULT = "POSITIVERESULT";
149     @Getter
150     private static final String NEGATIVE_RESULT = "NEGATIVERESULT";
151     private static final String SECTION_NAME = "TestsFile";
152 }
