32  /**
33   * Read a file containing the population data.
34   */
35  @Slf4j
36  public class PopulationsFileReader extends DataFileReader {

38      /**
39       * Create the populations file reader.
40       * @param populationFile the [xml] tag of the config file that is to be read.
41       * @param dbImpl         the implementation of the database object.
42       */
43      public PopulationsFileReader(final DataFiles.PopulationFile populationFile,
44                                   final DatabaseImpl dbImpl) {
45  
46          super();
47          this.database = dbImpl;
48          this.dataFile = populationFile.getName();
49          this.dateFields = new HashSet<>();
50          this.dateFormat = populationFile.getDateFormat();
51          this.insertedColInfo = new TreeMap<>();
52          this.populationFile = populationFile;
53  
54          if (populationFile.getLifeHistory() != null) {
55              readLifeHistory();
56          } else if (populationFile.getPopulation() != null) {
57              readPopulation();
58          } else {
59              throw new BroadwickException("Cannot read Populations section - No lifeHistory or population section found.");
60          }
61      }

63      /**
64       * Read the life history declarations of a population file section in the configuration file, if there are errors,
65       * they are noted and and exception is thrown with ALL the errors at the end of the method.
66       */
67      private void readLifeHistory() {
68          tableName = LIFE_HISTORIES_TABLE_NAME;
69          final Collection<String> primaryKeys = Arrays.asList(ID);
70          final StringBuilder errors = new StringBuilder();
71          this.createTableCommand = new StringBuilder();
72          updateCreateTableCommand(ID, populationFile.getLifeHistory().getIdColumn(), " VARCHAR(128) NOT NULL, ",
73                                   insertedColInfo, createTableCommand, LIFE_HISTORIES_TABLE_NAME, SECTION_NAME, errors);
74          updateCreateTableCommand(DATE_OF_BIRTH, populationFile.getLifeHistory().getDateOfBirthColumn()," INT, ", 
75                                   insertedColInfo, createTableCommand, LIFE_HISTORIES_TABLE_NAME,
76                                   SECTION_NAME, errors);
77          updateCreateTableCommand(LOCATION_OF_BIRTH, populationFile.getLifeHistory().getLocationOfBirthColumn(),
78                                   " VARCHAR(128), ", insertedColInfo, createTableCommand, LIFE_HISTORIES_TABLE_NAME,
79                                   SECTION_NAME, errors);
80          updateCreateTableCommand(DATE_OF_DEATH, populationFile.getLifeHistory().getDateOfDeathColumn(), " INT, ",
81                                   insertedColInfo, createTableCommand, LIFE_HISTORIES_TABLE_NAME, SECTION_NAME, errors);
82          updateCreateTableCommand(LOCATION_OF_DEATH, populationFile.getLifeHistory().getLocationOfDeathColumn(),
83                                   " VARCHAR(128), ", insertedColInfo, createTableCommand, LIFE_HISTORIES_TABLE_NAME,
84                                   SECTION_NAME, errors);
85          updateCreateTableCommand(SPECIES, populationFile.getLifeHistory().getSpeciesColumn(),
86                                   " VARCHAR(32), ", insertedColInfo, createTableCommand, LIFE_HISTORIES_TABLE_NAME,
87                                   SECTION_NAME, errors);
88  
89          if (populationFile.getCustomTags() != null) {
90              for (CustomTags.CustomTag tag : populationFile.getCustomTags().getCustomTag()) {
91                  updateCreateTableCommand(tag.getName(), tag.getColumn(), " VARCHAR(128), ",
92                                           insertedColInfo, createTableCommand, LIFE_HISTORIES_TABLE_NAME, SECTION_NAME, errors);
93                  if ("date".equals(tag.getType())) {
94                      dateFields.add(tag.getColumn());
95                  }
96              }
97          }
98  
99          if (populationFile.getLifeHistory().getDateOfBirthColumn() > 0) {
100             dateFields.add(populationFile.getLifeHistory().getDateOfBirthColumn());
101         }
102 
103         if (populationFile.getLifeHistory().getDateOfDeathColumn() > 0) {
104             dateFields.add(populationFile.getLifeHistory().getDateOfDeathColumn());
105         }
106 
107         final StringBuilder createIndexCommand = new StringBuilder();
108         createTableCommand.append("PRIMARY KEY (").append(asCsv(primaryKeys)).append("));");
109 
110 //        createIndexCommand.append(" DROP INDEX IDX_POP_ID IF EXISTS;");
111         createIndexCommand.append(String.format(" CREATE INDEX IDX_POP_ID ON %s (%s);",
112                                                 LIFE_HISTORIES_TABLE_NAME, ID));
113 //        createIndexCommand.append(" DROP INDEX IDX_POP_DATE IF EXISTS;");
114         createIndexCommand.append(String.format(" CREATE INDEX IDX_POP_DATE ON %s (%s,%s,%s);",
115                                                 LIFE_HISTORIES_TABLE_NAME, ID, DATE_OF_BIRTH, DATE_OF_DEATH));
116 
117         createTableCommand.append(createIndexCommand.toString());
118 
119         insertString = String.format("MERGE INTO %s (%s) KEY (%S) VALUES (%s)",
120                                      LIFE_HISTORIES_TABLE_NAME, 
121                                      asCsv(insertedColInfo.keySet()), 
122                                      asCsv(primaryKeys), 
123                                      asQuestionCsv(insertedColInfo.keySet()));
124 
125         if (errors.length() > 0) {
126             log.error(errors.toString());
127             throw new BroadwickException(errors.toString());
128         }
129     }

131     /**
132      * Read the populations declarations of a population file section in the configuration file, if there are errors,
133      * they are noted and and exception is thrown with ALL the errors at the end of the method.
134      */
135     private void readPopulation() {
136         tableName = POPULATIONS_TABLE_NAME;
137         final StringBuilder errors = new StringBuilder();
138         this.createTableCommand = new StringBuilder();
139         updateCreateTableCommand(LOCATION, populationFile.getPopulation().getLocationIdColumn(), " VARCHAR(128), ",
140                                  insertedColInfo, createTableCommand, POPULATIONS_TABLE_NAME, SECTION_NAME, errors);
141         updateCreateTableCommand(POPULATION, populationFile.getPopulation().getPopulationSizeColumn(), " VARCHAR(128), ",
142                                  insertedColInfo, createTableCommand, POPULATIONS_TABLE_NAME, SECTION_NAME, errors);
143         updateCreateTableCommand(DATE_LC, populationFile.getPopulation().getPopulationDateColumn(), " INT, ",
144                                  insertedColInfo, createTableCommand, POPULATIONS_TABLE_NAME, SECTION_NAME, errors);
145 
146 
147         dateFields.add(populationFile.getPopulation().getPopulationDateColumn());
148 
149         if (populationFile.getPopulation().getSpeciesColumn() != null) {
150             if (populationFile.getPopulation().getSpeciesColumn() != 0) {
151                 updateCreateTableCommand(SPECIES, populationFile.getPopulation().getSpeciesColumn(),
152                                          " VARCHAR(32), ", insertedColInfo, createTableCommand, POPULATIONS_TABLE_NAME,
153                                          SECTION_NAME, errors);
154             } else {
155                 errors.append("No ").append(SPECIES).append(" column set in ").append(SECTION_NAME).append(" section\n");
156             }
157         }
158 
159         if (populationFile.getCustomTags() != null) {
160             for (CustomTags.CustomTag tag : populationFile.getCustomTags().getCustomTag()) {
161                 if (populationFile.getPopulation().getSpeciesColumn() != 0) {
162                     updateCreateTableCommand(tag.getName(), tag.getColumn(), " VARCHAR(128), ",
163                                              insertedColInfo, createTableCommand, POPULATIONS_TABLE_NAME, SECTION_NAME, errors);
164                 } else {
165                     errors.append("No ").append(tag.getName()).append(" column set in ").append(SECTION_NAME).append(" section\n");
166                 }
167             }
168         }
169 
170         if (populationFile.getPopulation().getPopulationDateColumn() > 0) {
171             dateFields.add(populationFile.getPopulation().getPopulationDateColumn());
172         }
173 
174         final StringBuilder createIndexCommand = new StringBuilder();
175         createTableCommand.deleteCharAt(createTableCommand.length() - 1);
176         createTableCommand.append(");");
177 
178 //        createIndexCommand.append(" DROP INDEX IDX_POP_LOCATION IF EXISTS;");
179         createIndexCommand.append(String.format(" CREATE INDEX IDX_POP_LOCATION ON %s (%s);",
180                                                 POPULATIONS_TABLE_NAME, LOCATION));
181 
182         createTableCommand.append(createIndexCommand.toString());
183 
184         insertString = String.format("INSERT INTO %s (%s) VALUES (%s)",
185                                      POPULATIONS_TABLE_NAME, asCsv(insertedColInfo.keySet()), asQuestionCsv(insertedColInfo.keySet()));
186 
187         if (errors.length() > 0) {
188             log.error(errors.toString());
189             throw new BroadwickException(errors.toString());
190         }
191     }

193     @Override
194     public final int insert() {
195         log.trace("PopulationFileReader insert");
196 
197         int inserted = 0;
198         try (Connection connection = database.getConnection()) {
199             createTable(tableName, createTableCommand.toString(), connection);
200 
201             inserted = insert(connection, tableName, insertString, dataFile, dateFormat, insertedColInfo, dateFields);
202         } catch (Exception ex) {
203             log.error("Error reading population data {}. {}", ex.getLocalizedMessage(), Throwables.getStackTraceAsString(ex));
204             throw new BroadwickException(ex);
205         }
206         return inserted;
207     }
