28  /**
29   * A subtype of Graph which is a (directed, rooted) tree. Actually what we have here is a rooted tree, that is, there is
30   * a designated single vertex, the root, from which we measure the shortest path to each vertex, which we call its
31   * depth; the maximum over all such depths is the tree's height.
32   * <p/>
33   * The implementation here is backed by the DelegateTree implementation in the JUNG library.
34   * @param <V> the vertex type.
35   * @param <E> the edge type.
36   */
37  @Slf4j
38  public class Tree<V extends Vertex, E extends Edge<V>> implements broadwick.graph.Graph<V, E>, Serializable {
39  
40      /**
41       * Creates an instance of the tree.
42       */
43      public Tree() {
44          tree = new DelegateTree<>();
45          vertexmaps = new HashMap<>();
46          edgemaps = new HashMap<>();
47      }
48  
49      /**
50       * Returns the root of this tree. The root is defined to be the vertex (designated either at the tree's creation
51       * time, or as the first vertex to be added) with respect to which vertex depth is measured.
52       * @return the root of this tree
53       */
54      public final V getRoot() {
55          return tree.getRoot();
56      }
57  
58      /**
59       * Returns the maximum depth in this tree.
60       * @return the maximum depth in this tree
61       */
62      public final int getHeight() {
63          return tree.getHeight();
64      }

66      /**
67       * Returns the (unweighted) distance of vertex from the root of this tree.
68       * @param vertex the vertex whose depth is to be returned.
69       * @return the length of the shortest unweighted path from vertex to the root of this tree
70       */
71      public final int getDepth(final V vertex) {
72          return tree.getDepth(vertex);
73      }

81      /**
82       * Obtain the sub-tree with <code>vertex</code> as it's root.
83       * @param vertex the root node of the subtree.
84       * @return a tree object that is a subtree of the current tree.
85       */
86      public final Tree<V, E> getSubTree(final V vertex) {
87          final Tree<V, E> subTree = new Tree<>();
88          try {
89              subTree.tree = (DelegateTree<V, E>) TreeUtils.getSubTree(tree, vertex);
90          } catch (InstantiationException | IllegalAccessException ex) {
91              log.error("Could not create subtree. {}", ex.getLocalizedMessage());
92          }
93          return subTree;
94      }

172     @Override
173     public final boolean addEdge(final E e, final V v1, final V v2, final EdgeType edgeType) {
174         edgemaps.put(e.getId(), e);
175         if (EdgeType.DIRECTED.equals(edgeType)) {
176             return tree.addEdge(e, v1, v2, edu.uci.ics.jung.graph.util.EdgeType.DIRECTED);
177         } else {
178             return tree.addEdge(e, v1, v2, edu.uci.ics.jung.graph.util.EdgeType.UNDIRECTED);
179         }

218     /**
219      * Get the <code>vertex</code> in this graph that is the source of the (single) edge that is incident on this
220      * vertex. A vertex on a tree has only one incident edge but several.
221      * @param vertex the vertex whose predecessor is to be returned
222      * @return a <code>Collection</code> view of the predecessors of <code>vertex</code> in this graph
223      */
224     public final V getPredecessor(final V vertex) {
225         final Collection<E> inEdges = tree.getInEdges(vertex);
226         if (inEdges.size() != 1) {
227             throw new IllegalArgumentException("Vertex in tree has incorrect number of in edges.");
228         }
229         return inEdges.iterator().next().source;
230     }
