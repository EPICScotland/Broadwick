16  package broadwick.statistics.distributions;

18  import broadwick.math.Vector;
19  import lombok.extern.slf4j.Slf4j;
20  import org.apache.commons.math3.random.Well44497b;
21  
22  /**
23   * Sample from a Multivariate Normal (Gaussian) Distribution, a generalization of the one-dimensional (univariate) 
24   * normal distribution to higher dimensions. This is a simple wrapper class for the Apache Commons Math 
25   * multivariate normal library.
26   * <p>
27   * References:</p><p>
28   * <ul>
29   * <li><a href="http://en.wikipedia.org/wiki/Multivariate_normal_distribution"> Multivariatenormal Distribution</a></li>
30   * </ul>
31   * </p>
32   */
33  @Slf4j
34  public class MultivariateNormalDistribution implements ContinuousMultivariateDistribution {
35  
36      /**
37       * Create a normal distribution with a given mean and standard deviation.
38       * @param means  the [mathematical] vector of means.
39       * @param covariances the covariance matrix.
40       */
41      public MultivariateNormalDistribution(final double[] means, final double[][] covariances) {
42          this.mnd = new org.apache.commons.math3.distribution.MultivariateNormalDistribution(
43                  new Well44497b(System.currentTimeMillis() * Thread.currentThread().getId()),
44                  means, covariances);
45          
46  //        Find any real matrix A such that A AT = Σ. When Σ is positive-definite, the Cholesky decomposition is typically used,                                                                                                      
47  //        Let z = (z1, …, zN)T be a vector whose components are N independent standard normal variates .
48  //        Let x be μ + Az. This has the desired distribution due to the affine transformation property.
49      }
50  
51      @Override
52      public final Vector sample() {
53          return new Vector(mnd.sample());
54      }

56      
57      org.apache.commons.math3.distribution.MultivariateNormalDistribution mnd;
58  }
