16  package broadwick;
17  
18  import lombok.Getter;
19  import org.joda.time.DateTime;
20  import org.joda.time.Days;
21  import org.joda.time.format.DateTimeFormat;
22  import org.joda.time.format.DateTimeFormatter;
23  
24  /**
25   * All the constants for the Broadwick framework are defined here.
26   */
27  public final class BroadwickConstants {
28  
29      /**
30       * Utility constructor (hidden).
31       */
32      private BroadwickConstants() {
33      }
34  
35      /**
36       * Convert a date object to an integer (number of days from a fixed start date, here 1/1/1900). All dates in the
37       * database are stored as integer values using this method.
38       * @param date the date object we are converting.
39       * @return the number of days from a fixed 'zero date'.
40       */
41      public static int getDate(final DateTime date) {
42          return Days.daysBetween(ZERO_DATE, date).getDays();
43      }
44  
45      /**
46       * Convert an integer (number of days from a fixed start date, here 1/1/1900) to a DateTime object. All dates in the
47       * database are stored as integer values, this method converts that integer to a DateTime object.
48       * @param date the integer date offset.
49       * @return the datetime object corresponding to the 'zero date' plus date.
50       */
51      public static DateTime toDate(final int date) {
52          return ZERO_DATE.plusDays(date);
53      }
54  
55      /**
56       * Convert a date object to an integer (number of days from a fixed start date, here 1/1/1900). All dates in the
57       * database are stored as integer values using this method.
58       * @param date       the date object we are converting.
59       * @param dateFormat the format the date is in when doing the conversion.
60       * @return the number of days from a fixed 'zero date'.
61       */
62      public static int getDate(final String date, final String dateFormat) {
63          final DateTimeFormatter pattern = DateTimeFormat.forPattern(dateFormat);
64          final DateTime dateTime = pattern.parseDateTime(date);
65          return BroadwickConstants.getDate(dateTime);
66      }
67  
68      @Getter
69      @SuppressWarnings("PMD.UnusedPrivateField")
70      private static final DateTime ZERO_DATE = new DateTime(1900, 1, 1, 0, 0);
71  }
