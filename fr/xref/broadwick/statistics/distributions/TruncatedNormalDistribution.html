16  package broadwick.statistics.distributions;
17  
18  import broadwick.rng.RNG;
19  import lombok.Getter;
20  
21  /**
22   * An implementation of the truncated normal distribution, i.e. a normal distribution whose value is bounded either
23   * above, below or both.
24   * <p>
25   * References:</p><p>
26   * <ul>
27   * <li><a href="http://mathworld.wolfram.com/NormalDistribution.html"> Normal Distribution</a></li>
28   * </ul>
29   */
30  public class TruncatedNormalDistribution implements ContinuousDistribution {
31  
32      /**
33       * Create an instance of the normal distribution distribution truncated to given limits.
34       * @param mean the mean of the normal distribution.
35       * @param sd   the standard deviation of the distribution.
36       * @param lb   the lower bound of the distribution, no values lower than this will be returned.
37       * @param ub   the upper bound of the distribution, no values higher1 than this will be returned.
38       */
39      public TruncatedNormalDistribution(final double mean, final double sd,
40                                         final double lb, final double ub) {
41  
42          if (mean == Double.NaN || sd == Double.NaN
43              || lb == Double.NaN || ub == Double.NaN) {
44              throw new IllegalArgumentException("Invalid argument: TruncatedNormalDistribution cannot take NaN as argument");
45          }
46  
47          if (lb > ub) {
48              throw new IllegalArgumentException("Invalid argument: lower bound greater than upper bound");
49          }
50  
51          this.mean = mean;
52          this.sd = sd;
53          this.lower = lb;
54          this.upper = ub;
55          this.generator = new RNG(RNG.Generator.Well19937c);
56      }
57  
58      @Override
59      public double sample() {
60  
61          // Sample using the method of C.P. Robert (doi: 10.1007/BF00143942, arXiv:0907.4010 [stat.CO])
62          double x = Double.NaN;
63          double rho = 0.0;
64          double u = generator.getDouble();
65  
66          while (u > rho) {
67              double z = generator.getDouble(lower, upper);
68              if (0 > lower && 0 < upper) {
69                  rho = Math.exp(-z * z / 2.0);
70              } else if (upper < 0) {
71                  rho = Math.exp(((upper * upper) - (z * z)) / 2.0);
72              } else if (0 < lower) {
73                  rho = Math.exp(((lower * lower) - (z * z)) / 2.0);
74              }
75  
76              u = generator.getDouble();
77              x = z;
78          }
79  
80          return x;
81      }
82  
83      public double rejectionSample() {
84          // use a simple rejection sampling
85          final Normal dist = new Normal(mean, sd);
86          double val = dist.sample();
87  
88          while (val < lower || val > upper) {
89              val = dist.sample();
90          }
91          return val;
92      }
93  
94      @Getter
95      private double mean;
96      @Getter
97      private double sd;
98      private double lower;
99      private double upper;
100     private RNG generator;
101 }
