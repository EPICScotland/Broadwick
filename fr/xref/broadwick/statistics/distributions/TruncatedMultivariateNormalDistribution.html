18  import broadwick.math.Matrix;
19  import broadwick.math.Vector;

21  /**
22   * Sample from a truncated Multivariate Normal (Gaussian) Distribution, i.e. a normal distribution whose value is
23   * bounded either above, below or both.
24   * <p>
25   * References:</p><p>
26   * <ul>
27   * <li><a href="http://en.wikipedia.org/wiki/Multivariate_normal_distribution"> Multivariatenormal Distribution</a></li>
28   * </ul>
29   * </p>
30   */
31  public class TruncatedMultivariateNormalDistribution implements ContinuousMultivariateDistribution {
32  
33      /**
34       * Create an instance of a multivariate distribution that is bounded.
35       * @param means       the [mathematical] vector of the means of each variable.
36       * @param covariances the [mathematical] matrix of the covariances of each variable.
37       * @param lb          the [mathematical] vector of the lower bounds of each variable.
38       * @param ub          the [mathematical] vector of the upper bounds of each variable.
39       */
40      public TruncatedMultivariateNormalDistribution(final Vector means, final Matrix covariances,
41                                                     final Vector lb, final Vector ub) {
42          
43          if (means.length() != lb.length() || lb.length() != ub.length() 
44              || ub.length() != covariances.rows() || covariances.rows() != covariances.columns()) {
45                  throw new IllegalArgumentException("The lengths of the input vectors must be equal and the covariances matirx must be square with the same size as the input vectors.");
46          }
47          
48          for (int i=0; i< means.length(); i++) {
49              if (means.element(i) < lb.element(i) || means.element(i) > ub.element(i)) {
50                  throw new IllegalArgumentException("The means of the distribution must lie between the lower and upper bounds");
51              }
52              
53              if (lb.element(i) > ub.element(i)) {
54                  throw new IllegalArgumentException("The lower bound of the distribution must be less than the upper bound");
55              }
56          }
57          
58          this.means = means;
59          this.covariances = covariances;
60          this.upperBounds = ub;
61          this.lowerBounds = lb;
62          this.n = means.toArray().length;
63      }
64  
65      @Override
66      public Vector sample() {
67  
68          Vector proposal = new Vector(n);
69  
70          // Gibbs sampler of Christian Robert (arxiv:0907.4010v1 [stat.CO])
71          // Robert, C.P, "Simulation of truncated normal variables",
72          //   Statistics and Computing, pp. 121-125 (1995).
73          Matrix covInv = covariances.inverse();
74          for (int i = 0; i < n; i++) {
75              // get the (n-1) vector from the i-th column of the covariances matrix, removing the i-th row.
76              Matrix sigmaI = new Matrix(n - 1, 1);
77              for (int j = 0; j < n - 1; j++) {
78                  if (j != i) {
79                      sigmaI.setEntry(j, 0, covariances.element(j, i));
80                  }
81              }
82  
83              // Get the inverse of the (n-1)(n-1) matrix obtained from the covariance matrix removing the
84              // ith row and column.
85              Matrix sigmaIinv = new Matrix(n - 1, n - 1);
86              for (int j = 0; j < n - 1; j++) {
87                  if (j != i) {
88                      for (int k = 0; k < n - 1; k++) {
89                          if (k != i) {
90                              sigmaIinv.setEntry(j, k, covInv.element(j, k));
91                          }
92                      }
93                  }
94              }
95  
96              // x_i is the (n-1) vector of components not being updated at this iteration.
97              Matrix xI = new Matrix(1, n - 1);
98              Matrix muI = new Matrix(1, n - 1);
99              for (int j = 0; j < n - 1; j++) {
100                 if (j != i) {
101                     xI.setEntry(0, j, means.element(j));
102                     muI.setEntry(0, j, means.element(j));
103                 }
104             }
105 
106             // mui is E(xi|x_i)
107             //  mui = mu(i) + sigmai_i * sigma_i_iInv * (x_i - mu_i);
108             Matrix diff = xI.transpose().subtract(muI.transpose());
109             double mui = means.element(i) + sigmaI.transpose().multiply(sigmaIinv).multiply(diff).element(0, 0);
110             double s2i = covariances.element(i, i) - sigmaI.transpose().multiply(sigmaIinv).multiply(sigmaI).element(0, 0);
111 
112             // now draw from the 1-d normal truncated to [lb, ub]
113             TruncatedNormalDistribution dist = new TruncatedNormalDistribution(mui, Math.sqrt(s2i),
114                                                                                lowerBounds.element(i),
115                                                                                upperBounds.element(i));
116             proposal.setEntry(i, dist.sample());
117 
118         }
119

122          boolean proposedStepOutOfBounds;

125          do {

127          proposal = mvn.sample();
128 
129          for (int i = 0; i < n; i++) {
130          if (proposal.element(i) < lowerBounds.element(i)
131          || proposal.element(i) > upperBounds.element(i)
132          || Math.abs(proposal.element(i)) < 0.00001) {
133          proposedStepOutOfBounds = true;
134          System.out.println(String.format("Failed to pick proposal [attempt %d]. %f<%f<%f", attempts,
135          lowerBounds.element(i), proposal.element(i), upperBounds.element(i)));
136          break;
137          }
138          }
139          attempts++;
140          } while (proposedStepOutOfBounds);
141          */
142         return proposal;
143     }

147     private final Vector upperBounds;
148     private final Vector means;
149     private final Matrix covariances;
