46      /**
47       * Create the object that will read the data element of the configuration file.
48       * @param data the <data/> element of the configuration file.
49       */
50      public DataReader(final Project.Data data) {
51          this.data = data;
52  
53          // if there is a data section in the config file let's read it.
54          if (data != null) {
55              if (data.getDatabases() != null) {
56                  // there is a database mentioned in the config file so let's use it
57                  dbName = data.getDatabases().getName();
58  //                dbImpl = new FirebirdDatabase(dbName, false);
59  //                dbImpl = new DerbyDatabase(dbName, false);
60  //                dbImpl = new HyperSqlDatabase(dbName, false);
61                  dbImpl = new H2Database(dbName, false);
62              } else if (data.getDatafiles() != null) {
63                  // there is a datafiles section in the config file so we will read them and create a randomly named 
64                  // database.
65                  final String db = "broadwick_" + RandomStringUtils.randomAlphanumeric(8);
66                  dbName = db + "_db";
67  //                dbImpl = new FirebirdDatabase(dbName, true);
68  //                dbImpl = new DerbyDatabase(dbName, true);
69  //                dbImpl = new HyperSqlDatabase(dbName, true);
70                  dbImpl = new H2Database(dbName, true);
71                  readDataSection();
72              } else {
73                  throw new BroadwickException("There is a <data> section in the configuration file but neither a <database> nor <datafiles> section.");
74              }
75  
76              lookup = new Lookup(dbImpl);
77              logDbStatistics();
78          }
79  
80      }
81  
82      /**
83       * Read the data section from the configuration file. If a database section is found then it is used and any files
84       * specified are ignored.
85       */
86      public final void readDataSection() {
87          final StopWatch sw = new StopWatch();
88          final DataFiles files = data.getDatafiles();
89          if (files != null) {
90              sw.start();
91              readDataFiles(files);
92              sw.stop();
93              log.info("Processed input data in {} ms", sw.toString());
94              log.info("Data stored internally in {}.", dbName);
95          }
96      }
97  
98      @Override
99      public final void close() {
100         if (dbImpl != null) {
101             log.trace("Closing database connection");
102             dbImpl.close();
103         }
104     }

106     /**
107      * Save a message giving the numbers of elements in the database.
108      */
109     private void logDbStatistics() {
110         log.info("Read {} population data from the database.", lookup.getNumAnimals());
111         log.info("Read {} locations data from the database.", lookup.getNumLocations());

116     /**
117      * Read the datafiles, each section type of the <datafiles/> element (e.g. all the <MovementFile/>s) are read in
118      * parallel and each individual element is its read in parallel with the data saved to an internal database. Once
119      * the datafiles have been read some post-processing is run to determine the locations of individuals at each time
120      * step throughout the simulation.
121      * @param files the Datafiles object from the configuration file.
122      */
123     private void readDataFiles(final DataFiles files) {
124 
125         try {
126             final String addingFileMsg = "Reading %s ...";
127 
128             readAllLocationSections(files.getLocationsFile(), addingFileMsg);
129             readAllPopulationSections(files.getPopulationFile(), addingFileMsg);
130             readAllTestSections(files.getTestsFile(), addingFileMsg);
131             readAllFullMovementSections(files.getFullMovementFile(), addingFileMsg);
132             readAllDirectedMovementSections(files.getDirectedMovementFile(), addingFileMsg);
133             readAllBatchedMovementSections(files.getBatchMovementFile(), addingFileMsg);
134         } catch (Exception e) {
135             log.error("Failure reading data file section. {}", e.getLocalizedMessage());
136             throw new BroadwickException(String.format("Failure reading data file section. %s", e.getLocalizedMessage()));
137         }
138     }

140     /**
141      * Add a description of a configuration file section to an internal key value pair so that the correct column of a
142      * data file can be read. Each <datafile> section in the configuration contains an element and the column location
143      * in the data file where this element can be found. This method update a key-value pair of the element name and
144      * location, recording errors as it finds them.
145      * @param elementName     the name of the element in the configuration file.
146      * @param indexInDataFile the location of the element in each row of the data file.
147      * @param keyValueMapping a table of the keyValueMapping name index and type.
148      * @param errors          a description of any parsing errors found.
149      * @param recordErrors    flag to control whether or not error are recorded in the errors parameter.
150      * @param sectionName     the name of the section from which the element being added to the table has been read.
151      */
152     protected final void updateSectionDefiniton(final String elementName, final int indexInDataFile,
153                                                 final Map<String, Integer> keyValueMapping, final StringBuilder errors,
154                                                 final boolean recordErrors, final String sectionName) {
155         if (indexInDataFile != 0) {
156             keyValueMapping.put(elementName, indexInDataFile);
157         } else {
158             if (recordErrors) {
159                 errors.append("No ").append(elementName).append(" column set in ").append(sectionName).append(" section\n");
160             }
161         }
162     }

164     /**
165      * Read the directed movements sections of the configuration file.
166      * @param directedMovementFiles a collection of directedMovement sections.
167      * @param addingFileMsg         formatted string (in the form "Adding %s to reading list") to be added to log files.
168      * @return the number of movements read.
169      */
170     private int readAllDirectedMovementSections(final List<DirectedMovementFile> directedMovementFiles, final String addingFileMsg) {
171         int elementsRead = 0;
172 
173         for (DataFiles.DirectedMovementFile file : directedMovementFiles) {
174             log.trace(String.format(addingFileMsg, file.getName()));
175             final DirectedMovementsFileReader movementsFileReader = new DirectedMovementsFileReader(file, dbImpl);
176             elementsRead += movementsFileReader.insert();
177         }
178 
179         if (elementsRead > 0) {
180             log.info("Read {} directed movements from {} files.", elementsRead, directedMovementFiles.size());
181         }
182         return elementsRead;
183     }
