16  package broadwick.graph.algorithms;
17  
18  import broadwick.graph.Edge;
19  import broadwick.graph.EdgeType;
20  import edu.uci.ics.jung.algorithms.shortestpath.DijkstraDistance;
21  import broadwick.graph.Graph;
22  import broadwick.graph.Vertex;
23  import edu.uci.ics.jung.algorithms.shortestpath.DijkstraShortestPath;
24  import java.util.List;
25  import org.apache.commons.collections15.Transformer;
26  
27  /**
28   * Calculates distances in a specified graph, using Dijkstra's single-source-shortest-path algorithm. All edge weights
29   * in the graph must be nonnegative; otherwise an
30   * <code>IllegalArgumentException</code> will be thrown.
31   * @param <V> the vertex type.
32   * @param <E> the edge type.
33   */
34  public class ShortestPath<V extends Vertex, E extends Edge<V>> {
35  
36      /**
37       * Create a ShortestPath instance.
38       * @param graph the network/graph on which we will be looking for shortest paths.
39       */
40      public ShortestPath(final Graph<V, E> graph) {
41  
42          if (EdgeType.DIRECTED.equals(graph.getEdgeType())) {
43              jungGraph = new edu.uci.ics.jung.graph.DirectedSparseMultigraph<>();
44          } else if (EdgeType.UNDIRECTED.equals(graph.getEdgeType())) {
45              jungGraph = new edu.uci.ics.jung.graph.UndirectedSparseMultigraph<>();
46          } else {
47              throw new IllegalArgumentException("Could not create ShortestPath object for unknown grpah type.");
48          }
49          for (E edge : graph.getEdges()) {
50              jungGraph.addEdge(edge, edge.getSource(), edge.getDestination());
51          }
52  
53  
54          weightTransformer = new EdgeWeightTransformer();
55      }
56  
57      /**
58       * Returns the length of a shortest path from the source to the target vertex, or null if the target is not
59       * reachable from the source. If either vertex is not in the graph for which this instance was created, throws
60       * <code>IllegalArgumentException</code>.
61       * @param source the source
62       * @param target the target
63       * @return the distance from source to target
64       */
65      public final double calculateDistance(final V source, final V target) {
66  
67          final DijkstraDistance<V, E> distance = new DijkstraDistance<>(jungGraph, weightTransformer);
68          final Number d = distance.getDistance(source, target);
69          return d == null ? 0 : d.doubleValue();
70      }
71  
72      /**
73       * Returns a List of the edges on the shortest path from source to target, in order of their occurrence on this
74       * path. If either vertex is not in the graph for which this instance was created, throws IllegalArgumentException
75       * @param source the vertex from which distances are measured
76       * @param target the number of vertics for which to measure distances
77       * @return a list of edges that comprise the shortest path from source to target.
78       */
79      public final List<E> getEdgesInPath(final V source, final V target) {
80          final DijkstraShortestPath<V,E> path = new DijkstraShortestPath<>(jungGraph);
81          return path.getPath(source, target);
82      }
83      private edu.uci.ics.jung.graph.AbstractTypedGraph<V,E> jungGraph;
84      private final Transformer<E, Number> weightTransformer;
85  
86      /**
87       * Transformer class to transform the edge of a graph to a double (it's weight).
88       */
89      private class EdgeWeightTransformer implements Transformer<E, Number> {
90  
91          @Override
92          public Number transform(final E edge) {
93              // all nonnull values are instances of broadwick.graph.Edge
94              if (edge != null) {
95                  return ((Edge) edge).getWeight();
96              } else {
97                  return 1.0;
98              }
99          }
100     }
