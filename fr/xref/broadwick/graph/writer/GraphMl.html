33  /**
34   * Simple class to write the nodes/edges of a network in graphML format.
35   */
36  public final class GraphMl {
37  
38      /**
39       * hidden constructor for static class.
40       */
41      private GraphMl() {
42          // no constructor
43      }
44  
45      /**
46       * Get the string representation of the network.
47       * @param network  the network object to be written.
48       * @param directed a boolean flag, true if the network is directed.
49       * @return a string representing a document.
50       */
51      public static String toString(final Graph<? extends Vertex, ? extends Edge<?>> network, final boolean directed) {
52          // graphml document header
53          final Element graphml = new Element("graphml", "http://graphml.graphdrawing.org/xmlns");
54          final Document document = new Document(graphml);
55          final Namespace xsi = Namespace.getNamespace("xsi", "http://www.w3.org/2001/XMLSchema-instance");
56          final Namespace schemLocation = Namespace.getNamespace("schemLocation", "http://graphml.graphdrawing.org/xmlns http://graphml.graphdrawing.org/xmlns/1.0rc/graphml.xsd");
57  
58          // add Namespace
59          graphml.addNamespaceDeclaration(xsi);
60          graphml.addNamespaceDeclaration(schemLocation);
61  
62          // keys for graphic representation
63          for (VertexAttribute attr : network.getVertexAttributes()) {
64              final Element element = new Element("key");
65              element.setAttribute("id", attr.getName());
66              element.setAttribute("for", "node");
67              element.setAttribute("attr.name", attr.getName());
68              element.setAttribute("attr.type", attr.getType().getName());
69              if (attr.getDefaultValue() != null) {
70                  final Element defaultValueElement = new Element("default");
71                  defaultValueElement.addContent(attr.getDefaultValue().toString());
72                  element.addContent(defaultValueElement);
73              }
74              graphml.addContent(element);
75          }
76  
77          for (EdgeAttribute attr : network.getEdgeAttributes()) {
78              final Element element = new Element("key");
79              element.setAttribute("id", attr.getName());
80              element.setAttribute("for", "edge");
81              element.setAttribute("attr.name", attr.getName());
82              element.setAttribute("attr.type", attr.getType().getName());
83              if (attr.getDefaultValue() != null) {
84                  final Element defaultValueElement = new Element("default");
85                  defaultValueElement.addContent(attr.getDefaultValue());
86                  element.addContent(defaultValueElement);
87              }
88              graphml.addContent(element);
89          }
90  
91          final Element graph = new Element("graph");
92          graph.setAttribute("id", "G");
93          if (directed) {
94              graph.setAttribute("edgedefault", "directed");
95          } else {
96              graph.setAttribute("edgedefault", "undirected");
97          }
98          graphml.addContent(graph);
99  
100         final ImmutableList<Vertex> vertices = ImmutableList.copyOf(network.getVertices());
101         final Iterator<Vertex> vertexIterator = vertices.iterator();
102         while (vertexIterator.hasNext()) {
103             final Vertex vertex = vertexIterator.next();
104             addNode(vertex, graph);
105         }
106 
107         final ImmutableList<Edge<? extends Vertex>> edges;
108         edges = (ImmutableList<Edge<? extends Vertex>>) ImmutableList.copyOf(network.getEdges());
109         final UnmodifiableIterator<Edge<? extends Vertex>> edgeIterator = edges.iterator();
110         while (edgeIterator.hasNext()) {
111             addEdge(edgeIterator.next(), graph);
112         }
113 
114         final XMLOutputter outputter = new XMLOutputter(Format.getPrettyFormat());
115         return outputter.outputString(document).replaceAll("xmlns=\"\" ", "");
116     }
117 
118     /**
119      * Get the string representation of the network.
120      * @param network the network object to be written.
121      * @return a string representing a document.
122      */
123     public static String toString(final Graph<? extends Vertex, ? extends Edge<?>> network) {
124         return GraphMl.toString(network, false);
125     }
126 
127     /**
128      * Save the graph as in graphML format in the given file.
129      * @param network the network object to be saved.
130      * @param file    the name of the file in which the graph will be saved.
131      */
132     public static void save(final String file, final Graph<? extends Vertex, ? extends Edge<Vertex>> network) {
133         try (FileOutput fo = new FileOutput(file)) {
134             fo.write(GraphMl.toString(network));
135         }
136     }

138     /**
139      * Add an edge to the graphML document.
140      * @param edge    the edge to be added to the graph element.
141      * @param element the graph element of the graphML document
142      */
143     private static void addEdge(final Edge edge, final Element element) {
144 
145         final String id = edge.getId();
146 
147         final String source = edge.getSource().getId();
148         final String target = edge.getDestination().getId();
149 
150         final Element elem = new Element("edge");
151         elem.setAttribute("id", "e" + id);
152         elem.setAttribute("source", source);
153         elem.setAttribute("target", target);
154 
155         for (final Iterator it = edge.getAttributes().iterator(); it.hasNext();) {
156             final EdgeAttribute attr = (EdgeAttribute) it.next();
157             final Element data = new Element("data");
158             data.setAttribute("key", attr.getName());
159             data.setText(attr.getValue());
160             elem.addContent(data);
161         }
162 
163         element.addContent(elem);
164     }
165 
166     /**
167      * Add a node to the graphML document.
168      * @param vertex  the id of the node.
169      * @param element the graph element of the graphML document.
170      */
171     private static void addNode(final Vertex vertex, final Element element) {
172         final Element node = new Element("node");
173         node.setAttribute("id", vertex.getId());
174 
175         for (VertexAttribute attr : vertex.getAttributes()) {
176             final Element data = new Element("data");
177             data.setAttribute("key", attr.getName());
178             data.setText(attr.getValue().toString());
179             node.addContent(data);
180         }
181 
182         element.addContent(node);
183     }
